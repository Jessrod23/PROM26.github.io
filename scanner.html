async function verificarQR(id){ const resultado = document.getElementById("resultadoScan"); const docRef = db.collection("boletos").doc(id); const docSnap = await docRef.get(); // ====================================== // QR INVÁLIDO // ====================================== if(!docSnap.exists){ resultado.innerHTML = `
QR INVÁLIDO ❌
`; alert("QR INVÁLIDO ❌"); return; } const data = docSnap.data(); // ====================================== // OBTENER NOMBRE // ====================================== let nombre = "SIN REGISTRAR"; if(data.nombre){ nombre = data.nombre; } // ====================================== // QR CLONADO // ====================================== if(data.usado){ resultado.innerHTML = `
QR CLONADO ⚠️
${nombre}
${id}
`; alert(` QR CLONADO ⚠️ NOMBRE: ${nombre} QR: ${id} `); return; } // ====================================== // ACTUALIZAR // ====================================== await docRef.update({ usado:true, horaEntrada:new Date() }); // ====================================== // MOSTRAR // ====================================== resultado.innerHTML = `
ACCESO PERMITIDO ✅
${nombre}
${id}
${data.tipo}
`; // ====================================== // ALERTA // ====================================== alert(` ACCESO PERMITIDO ✅ NOMBRE: ${nombre} QR: ${id} TIPO: ${data.tipo} `); }
